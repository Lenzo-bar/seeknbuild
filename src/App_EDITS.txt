  // After YES dialog: when the follow-up search finishes, generate new topic-specific
  // filters from the new results and APPEND them below the existing sidebar filters.
  // This implements the "Initial Filters + Additional Filters" two-block design.
  useEffect(() => {
    if (!pendingAppendQuery) return
    // Wait for all active searches to finish
    if (isSearching || isLlmSearching || isWebOnlySearching || isUrlAnalyzing || isAnalyzing) return

    const q = pendingAppendQuery
    setPendingAppendQuery(null)

    // Gather card samples from whichever tab was just searched
    const samples: {title:string; tags:string[]; source?:string}[] = [
      ...webCards.filter(c => (c as any).rank !== 0).slice(0, 8).map(c => ({ title: c.title, tags: (c as any).tags ?? [], source: c.source })),
      ...webOnlyCards.filter(c => (c as any).rank !== 0).slice(0, 8).map(c => ({ title: c.title, tags: (c as any).tags ?? [], source: c.source })),
      ...urlCards.filter(c => (c as any).rank !== 0).slice(0, 6).map(c => ({ title: c.title, tags: (c as any).tags ?? [] })),
    ].slice(0, 12)

    if (samples.length === 0) return

    // Call the API to generate new filters for the extended/new prompt
    fetch('/api/llm', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        query: `You are generating sidebar filter controls for a research app called SeekNBuild.

The user extended their search with: "${q}"

The new result cards returned were:
${samples.map((c,i) => `${i+1}. "${c.title}"${c.tags?.length ? ' [' + c.tags.slice(0,3).join(', ') + ']' : ''}`).join('\n')}

Generate 2-4 NEW sidebar filter sections for these additional results.
IMPORTANT: These filters will be shown in a separate "Additional Filters" block
BELOW the existing initial filters — so do NOT duplicate filter types already
common to the initial search (like Date, Difficulty). Focus on what's NEW or
DIFFERENT about this extended query compared to the original.

Option values MUST be words/phrases that appear in the card titles, snippets, or tags above.

Respond with ONLY a JSON array:
[{"id":"snake_id","label":"Short Label","type":"checkboxes","options":["A","B","C"]}]`,
        sidebarMode: true,
      }),
    })
    .then(r => r.json())
    .then(data => {
      const filters = Array.isArray(data.sidebarFilters) ? data.sidebarFilters : []
      if (filters.length > 0) appendSidebarFilters(filters)
    })
    .catch(() => {/* fail silently */})
  }, [pendingAppendQuery, isSearching, isLlmSearching, isWebOnlySearching, isUrlAnalyzing, isAnalyzing])
